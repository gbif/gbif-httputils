153   /**
154    * Creates a url form encoded http entity suitable for POST requests with a single given parameter
155    * encoded in utf8.
156    * 
157    * @param kvp the parameter map to encode
158    */
159   public static HttpEntity map2Entity(Map<String, String> kvp) {
160     List<NameValuePair> formparams = new ArrayList<NameValuePair>(kvp.size());
161     for (Map.Entry<String, String> entry : kvp.entrySet()) {
162       formparams.add(new BasicNameValuePair(entry.getKey(), entry.getValue()));
163     }
164     try {
165       return new UrlEncodedFormEntity(formparams, UTF_8);
166     } catch (UnsupportedEncodingException e) {
167       LOG.error("Can't encode post entity with utf8", e);
168     }
169     return null;
170   }

190   /**
191    * This creates a new threadsafe, multithreaded http client with support for http and https.
192    * Default http client values are partially overriden to use UTF8 as the default charset and an explicit timeout
193    * is required for configuration.
194    * 
195    * @param timeout in milliseconds
196    * @param maxConnections maximum allowed connections in total
197    * @param maxPerRoute maximum allowed connections per route
198    */
199   public static DefaultHttpClient newMultithreadedClient(int timeout, int maxConnections, int maxPerRoute) {
200     HttpParams params = new BasicHttpParams();
201     params.setParameter(CoreProtocolPNames.HTTP_CONTENT_CHARSET, UTF_8);
202     HttpConnectionParams.setConnectionTimeout(params, timeout);
203     HttpConnectionParams.setSoTimeout(params, timeout);
204     params.setLongParameter(ClientPNames.CONN_MANAGER_TIMEOUT, timeout);
205 
206     SchemeRegistry schemeRegistry = new SchemeRegistry();
207     schemeRegistry.register(new Scheme(HTTP_PROTOCOL, HTTP_PORT, PlainSocketFactory.getSocketFactory()));
208     schemeRegistry.register(new Scheme(HTTPS_PROTOCOL, HTTPS_PORT, SSLSocketFactory.getSocketFactory()));
209 
210     PoolingClientConnectionManager connectionManager = new PoolingClientConnectionManager(schemeRegistry);
211     connectionManager.setMaxTotal(maxConnections);
212     connectionManager.setDefaultMaxPerRoute(maxPerRoute);
213     return new DefaultHttpClient(connectionManager, params);
214   }

217   /**
218    * Parses a RFC2616 compliant date string such as used in http headers.
219    * 
220    * @see <a href="http://tools.ietf.org/html/rfc2616#section-3.3">RFC 2616</a> specification.
221    *      example:
222    *      Wed, 21 Jul 2010 22:37:31 GMT
223    * @param rfcDate RFC2616 compliant date string
224    * @return the parsed date or null if it cannot be parsed
225    */
226   public static Date parseHeaderDate(String rfcDate) {
227     try {
228       if (rfcDate != null) {
229         // as its not thread safe we create a new instance each time
230         return new SimpleDateFormat("EEE, dd MMM yyyy HH:mm:ss Z", Locale.US).parse(rfcDate);
231       }
232     } catch (ParseException e) {
233       LOG.debug("Can't parse RFC2616 date");
234     }
235     return null;
236   }

238   public static String responseAsString(HttpResponse response) {
239     String content = null;
240     HttpEntity entity = response.getEntity();
241     if (entity != null) {
242       try {
243         content = EntityUtils.toString(entity);
244         EntityUtils.consume(entity);
245       } catch (org.apache.http.ParseException e) {
246         LOG.error("ParseException consuming http response into string", e);
247       } catch (IOException e) {
248         LOG.error("IOException consuming http response into string", e);
249       }
250     }
251     return content;
252   }

274   private HttpContext buildContext(String uri, UsernamePasswordCredentials credentials) throws URISyntaxException {
275     HttpContext authContext = new BasicHttpContext();
276     if (credentials != null) {
277       URI authUri = new URI(uri);
278       AuthScope scope = new AuthScope(authUri.getHost(), AuthScope.ANY_PORT, AuthScope.ANY_REALM);
279 
280       CredentialsProvider credsProvider = new BasicCredentialsProvider();
281       credsProvider.setCredentials(scope, credentials);
282 
283       authContext.setAttribute(HttpClientContext.CREDS_PROVIDER, credsProvider);
284     }
285 
286     return authContext;
287   }

328   public StatusLine download(URL url, File downloadTo) throws IOException {
329     HttpGet get = new HttpGet(url.toString());
330 
331     // execute
332     HttpResponse response = client.execute(get);
333     final StatusLine status = response.getStatusLine();
334     // write to file only when download succeeds
335     if (success(status)) {
336       HttpEntity entity = response.getEntity();
337       if (entity != null) {
338         // copy stream to local file
339         FileUtils.forceMkdir(downloadTo.getParentFile());
340         OutputStream fos = new FileOutputStream(downloadTo, false);
341         try {
342           entity.writeTo(fos);
343         } finally {
344           fos.close();
345         }
346       }
347 
348       LOG.debug("Successfully downloaded {} to {}", url, downloadTo.getAbsolutePath());
349     } else {
350       LOG.error("Downloading {} to {} failed!: {}", url, downloadTo.getAbsolutePath(), status.getStatusCode());
351     }
352     return status;
353   }

359     Map<String, String> header = new HashMap<String, String>(1);
360     header.put(MODIFIED_SINCE, DateFormatUtils.SMTP_DATETIME_FORMAT.format(lastModified));
361 
362     try {
363       Response resp = get(url.toString(), header, null);
364       if (resp.getStatusCode() == HttpStatus.SC_NOT_MODIFIED) {
365         LOG.debug("Content not modified since last request");
366       }
367       return resp.content;
368     } catch (URISyntaxException e) {
369       LOG.error("Invalid URL provided: {}", url, e);
370     }
371     return null;
372   }

396   /**
397    * Downloads a url to a file if its modified since the date given.
398    * Updates the last modified file property to reflect the last servers modified http header.
399    * 
400    * @param downloadTo file to download to
401    * @return true if changed or false if unmodified since lastModified
402    */
403   public StatusLine downloadIfModifiedSince(final URL url, final Date lastModified, final File downloadTo)
404     throws IOException {
405 
406     HttpGet get = new HttpGet(url.toString());
407 
408     // prepare conditional GET request headers
409     if (lastModified != null) {
410       // DateFormatUtils is threadsafe
411       get.addHeader(MODIFIED_SINCE, DateFormatUtils.SMTP_DATETIME_FORMAT.format(lastModified));
412       LOG.debug("Conditional GET: {}", DateFormatUtils.SMTP_DATETIME_FORMAT.format(lastModified));
413     }
414 
415     // execute
416     HttpResponse response = client.execute(get);
417     final StatusLine status = response.getStatusLine();
418     if (status.getStatusCode() == HttpStatus.SC_NOT_MODIFIED) {
419       LOG.debug("Content not modified since last request");
420     }
421     // write to file only when download succeeds
422     else if (success(status)) {
423       HttpEntity entity = response.getEntity();
424       if (entity != null) {
425 
426         Date serverModified = null;
427         Header modHeader = response.getFirstHeader(LAST_MODIFIED);
428         if (modHeader != null) {
429           serverModified = parseHeaderDate(modHeader.getValue());
430         }
431 
432         // copy stream to local file
433         FileUtils.forceMkdir(downloadTo.getParentFile());
434         OutputStream fos = new FileOutputStream(downloadTo, false);
435         try {
436           entity.writeTo(fos);
437         } finally {
438           fos.close();
439         }
440         // update last modified of file with http header date from server
441         if (serverModified != null) {
442           downloadTo.setLastModified(serverModified.getTime());
443         }
444       }
445 
446       // close http connection
447       EntityUtils.consume(entity);
448 
449       LOG.debug("Successfully downloaded {} to {}", url, downloadTo.getAbsolutePath());
450     } else {
451       LOG.error("Downloading {} to {} failed!: {}", url, downloadTo.getAbsolutePath(), status.getStatusCode());
452     }
453 
454     return status;
455   }

458    * Downloads a url to a local file using conditional GET, i.e. only downloading the file again if it has been changed
459    * on the filesystem since the last download.
460    * 
461    * @param url url to download
462    * @param downloadTo file to download into and used to get the last modified date from
463    */
464   public StatusLine downloadIfModifiedSince(final URL url, final File downloadTo) throws IOException {
465     Date lastModified = null;
466     if (downloadTo.exists()) {
467       lastModified = new Date(downloadTo.lastModified());
468     }
469     return downloadIfModifiedSince(url, lastModified, downloadTo);
470   }

472   public HttpResponse executeGetWithTimeout(HttpGet get, int timeout) throws ClientProtocolException, IOException {
473     HttpParams httpParams = client.getParams();
474     // keep old values to rest afterwards
475     int ct = HttpConnectionParams.getConnectionTimeout(httpParams);
476     int st = HttpConnectionParams.getSoTimeout(httpParams);
477 
478     HttpConnectionParams.setConnectionTimeout(httpParams, timeout);
479     HttpConnectionParams.setSoTimeout(httpParams, timeout);
480 
481     HttpResponse response = null;
482     try {
483       response = client.execute(get);
484     } finally {
485       // rest to previous values
486       HttpConnectionParams.setConnectionTimeout(httpParams, ct);
487       HttpConnectionParams.setSoTimeout(httpParams, st);
488     }
489 
490     return response;
491   }

500   public Response get(String url, Map<String, String> headers, UsernamePasswordCredentials credentials)
501     throws IOException, URISyntaxException {
502     HttpGet get = new HttpGet(url);
503     // http header
504     if (headers != null) {
505       for (Map.Entry<String, String> header : headers.entrySet()) {
506         get.addHeader(StringUtils.trimToEmpty(header.getKey()), StringUtils.trimToEmpty(header.getValue()));
507       }
508     }
509     // authentication
510     HttpContext authContext = buildContext(url, credentials);
511     HttpResponse response = client.execute(get, authContext);
512 
513     Response result = new Response(response);
514     HttpEntity entity = response.getEntity();
515     if (entity != null) {
516       // Adding a default charset in case it is not found
517       result.content = EntityUtils.toString(entity, UTF_8);
518       EntityUtils.consume(entity);
519     }
520     return result;
521   }

549   public Response post(String uri, Map<String, String> headers, UsernamePasswordCredentials credentials,
550     HttpEntity requestEntity) throws IOException, URISyntaxException {
551     HttpPost post = new HttpPost(uri);
552     if (headers != null) {
553       for (Map.Entry<String, String> header : headers.entrySet()) {
554         post.addHeader(StringUtils.trimToEmpty(header.getKey()), StringUtils.trimToEmpty(header.getValue()));
555       }
556     }
557     if (requestEntity != null) {
558       post.setEntity(requestEntity);
559     }
560     // authentication
561     HttpContext authContext = buildContext(uri, credentials);
562     HttpResponse response = client.execute(post, authContext);
563 
564     // response
565     if (response != null) {
566       Response result = new Response(response);
567       HttpEntity respEntity = response.getEntity();
568       if (respEntity != null) {
569         result.content = EntityUtils.toString(respEntity);
570         EntityUtils.consume(respEntity);
571       }
572       return result;
573     }
574     return null;
575   }
