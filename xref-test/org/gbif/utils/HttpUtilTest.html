22      HttpUtil util = new HttpUtil(new DefaultHttpClient());
23      File tmp = File.createTempFile("httputils", "test");
24      tmp.deleteOnExit();
25      // a redirect to http://rs.gbif.org/extension/gbif/1.0/distribution.xml

36    /**
37     * Tests a condition get against an apache http server within GBIF.
38     * If the file being tested against has been changed in the last 24h this test is expected to fail!
39     * 
40     * @see <a href="http://dev.gbif.org/issues/browse/GBIFCOM-77">GBIFCOM-77</a>
41     */
42    @Test
43    public void testConditionalGet() throws ParseException, IOException {
44      DefaultHttpClient client = new DefaultHttpClient();
45      HttpUtil util = new HttpUtil(client);
46      // We know for sure it has changed since this date
47      Date last = HttpUtil.parseHeaderDate("Wed, 03 Aug 2009 22:37:31 GMT");
48      File tmp = File.createTempFile("vocab", ".xml");
49      URL url = new URL("http://rs.gbif.org/vocabulary/gbif/rank.xml");
50      assertTrue(util.downloadIfChanged(url, last, tmp));
51  
52      // Verify that it does not download with a conditional get of a day before "now"
53      // we dont use now because the apache server on the other side might have a slightly different clock and timezone
54      // configured that might lead to a mismatch.
55      // If the rank file was modified within the last 24h this test could fail though !!!
56      Date nearlyNow = new Date(System.currentTimeMillis() - 24 * 1000 * 60 * 60);
57      assertFalse(util.downloadIfChanged(url, nearlyNow, tmp));
58    }
